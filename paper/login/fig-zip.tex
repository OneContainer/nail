protozip = { 
  $files, $header transform zipdir($current)
  contents apply $header dir($files)
}
dir ($files) = {
  uint32 =  0x00034b50
  @count uint32
  files n_of @count {
    @off uint32
    @size uint32
    filename many (uint8 | ![0])
    $compr transform slice_u32($files @off @size)
    $decomp transform zlib($compr)
    contents apply $decomp (many uint8)
  }
}
